Procedimiento ronda(E/S: contricantes[tamano]: Entero, tamano: Entero);
Inicio
  Var i = 0, j = 0: Entero;
  Var tamano_ganadores: Entero;
  Var peleadores[2], ganadores[tamano_ganadores]: Entero;

  tamano_ganadores = tamano / 2;

  Si (tamano > 2) Entonces
    Mientras (i < tamano) Hacer;
      peleadores[1] = contricantes[i];
      peleadores[2] = contricantes[i + 1];

      ganadores[j] = batalla(peleadores);
      
      j = j + 1;
      i = i + 2;
    Fin_Mientras

    ronda(ganadores, tamano_ganadores);
    Sino 
      // Es la final
  Fin_Si
Fin_Procedimiento
