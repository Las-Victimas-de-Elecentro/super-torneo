Procedimiento Contrincantes (E/S: mínimo, máximo, I, J, Contrincantes[I]: Entero);
Inicio
	Num: Entero;
	máximo = 31;
	mínimo = 1;
	Para (I = 1 hasta I = 15) hacer; 
			Num = Aleatorio(mínimo, máximo);
			Contrincantes [I] = Num;
		Para (J = 1 hasta J = I) hacer;
			Si (Num = Contrincantes [J]) entonces;
				Num = Aleatorio(mínimo, máximo); 
			Sino
				Contrincantes [J] = Num;
			Fin_Si
		Fin_Para
	Fin_Para
Fin_Procedimiento

			Contador_Batallas2 = Contador_Batallas2 + 1;

			Batalla_Semis [K] = Resultados_Cuartos [Contador_Batallas3];
			Contador_Batallas3 = Contador_Batallas3 + 1;
			Enfrentamientos[K] = Batalla_Semis[K];

	Fin_Para
	Batallas (Ganador, Luchadores[I].ataque, Luchadores[I].defensa, Luchadores[I].velocidad, Luchadores[I].nombre);
	Resultado_Final = Ganador;
Fin_Procedimiento
